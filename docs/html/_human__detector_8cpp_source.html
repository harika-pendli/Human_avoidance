56 std::vector<std::string> HumanDetector::getOutputsNames(
57  const cv::dnn::Net& net) {
58  static std::vector<std::string> names;
59  if (names.empty()) {
60  // Get the indices of the output layers, i.e. the layers with unconnected
61  // outputs
62  std::vector<int> outLayers = net.getUnconnectedOutLayers();
63 
64  // get the names of all the layers in the network
65  std::vector<std::string> layersNames = net.getLayerNames();
66 
67  // Get the names of the output layers in names
68  names.resize(outLayers.size());
69  for (size_t i = 0; i < outLayers.size(); ++i)
70  names[i] = layersNames[outLayers[i] - 1];
71  }

75 void HumanDetector::rmOverlap(cv::Mat& frame, const std::vector<cv::Mat>& outs,
76  const std::vector<std::string>& classes) {
77  std::vector<int> classIds;
78  std::vector<float> confidences;
79  std::vector<cv::Rect> boxes;
80  Transform transform;
81 
82  for (size_t i = 0; i < outs.size(); ++i) {
83  // Scan through all the bounding boxes output from the network and keep only
84  // the ones with high confidence scores. Assign the box's class label as the
85  // class with the highest score for the box.
86  float* data = reinterpret_cast<float*>(outs[i].data);
87  for (int j = 0; j < outs[i].rows; ++j, data += outs[i].cols) {
88  cv::Mat scores = outs[i].row(j).colRange(5, outs[i].cols);
89  cv::Point classIdPoint;
90  double confidence;
91  // Get the value and location of the maximum score
92  minMaxLoc(scores, 0, &confidence, 0, &classIdPoint);
93  if (confidence > confidenceThreshold) {
94  int centerX = static_cast<int>(data[0] * frame.cols);
95  int centerY = static_cast<int>(data[1] * frame.rows);
96  int width = static_cast<int>(data[2] * frame.cols);
97  int height = static_cast<int>(data[3] * frame.rows);
98  int left = centerX - width / 2;
99  int top = centerY - height / 2;
100 
101  classIds.push_back(classIdPoint.x);
102  confidences.push_back(static_cast<float>(confidence));
103  boxes.push_back(cv::Rect(left, top, width, height));
104  }
105  }
106  }
107 
108  // Perform non maximum suppression to eliminate redundant overlapping boxes
109  // with lower confidences
110  std::vector<int> indices;
111  cv::dnn::NMSBoxes(boxes, confidences, confidenceThreshold, nmsThreshold,
112  indices);
113  unsigned int personid = 1;
114  for (auto idx : indices) {
115  cv::Rect box = boxes[idx];
116  float z_axis = transform.calculate_distance(box.height, frame.rows);
117  drawBox(classIds[idx], confidences[idx], box.x, box.y, box.x + box.width,
118  box.y + box.height, frame, classes, personid, z_axis);
119  transform.camera_robot_array(z_axis, box, frame);
120  std::cout << "Distance from person " << personid
121  << " is: " << z_axis << "m" << std::endl;
122  personid++;
123  }
124 }

126 void HumanDetector::detect(cv::CommandLineParser parser) {
127  std::vector<std::string> classes;
128 
129  ModelConfig yolov3;
130  DataReader input;
131  classes = yolov3.getClasses();
132 
133  cv::dnn::Net net =
134  cv::dnn::readNetFromDarknet(yolov3.getConfig(), yolov3.getWeights());
135  net.setPreferableBackend(cv::dnn::DNN_BACKEND_OPENCV);
136  net.setPreferableTarget(cv::dnn::DNN_TARGET_CPU);
137 
138  cv::VideoCapture cap;
139  cv::VideoWriter video;
140  cv::Mat frame, blob;
141 
142  try {
143  std::string input_Type = "";
144  std::string inputPath = input.getInput(parser, input_Type);
145  if (input_Type == "image") {
146  input.setImagepath(inputPath);
147  cap = input.imgProcessor('r', frame);
148  }
149  if (input_Type == "video") {
150  input.setVideopath(inputPath);
151  cap = input.videoProcessor('r', frame, video);
152  input.setOutputWidth(cap.get(cv::CAP_PROP_FRAME_WIDTH));
153  input.setOutputHeight(cap.get(cv::CAP_PROP_FRAME_HEIGHT));
154  video.open(
155  "../output/Result.mp4", cv::VideoWriter::fourcc('a', 'v', 'c', '1'),
156  24, cv::Size(input.getOutputWidth(), input.getOutputHeight()), true);
157  }
158  } catch (...) {
159  std::cout << "Could not open the input image/video stream" << std::endl;
160  }
161 
162  while (cv::waitKey(1) < 0) {
163  cap >> frame;
164 
165  // Stop the program if reached end of video
166  if (frame.empty()) {
167  std::cout << "Detection Completed" << std::endl;
168  std::cout << "Output file is stored in Output Folder" << std::endl;
169  cv::waitKey(3000);
170  break;
171  }
172  // Create a 4D blob from a frame.
173  cv::dnn::blobFromImage(frame, blob, 1 / 255.0,
174  cv::Size(inputWidth, inputHeight),
175  cv::Scalar(0, 0, 0), true, false);
176 
177  // Sets the input to the network
178  net.setInput(blob);
179 
180  // Runs the forward pass to get output of the output layers
181  std::vector<cv::Mat> outs;
182  net.forward(outs, getOutputsNames(net));
183 
184  // Remove the bounding boxes with low confidence
185  rmOverlap(frame, outs, classes);
186 
187  // Put efficiency information.
188  std::vector<double> layersTimes;
189  double freq = cv::getTickFrequency() / 1000;
190 
191  // The function getPerfProfile returns the overall time for inference(t)
192  // and the timings for each of the layers(in layersTimes)
193  double t = net.getPerfProfile(layersTimes) / freq;
194  std::string label = cv::format("Inference time for a frame : %.2f ms", t);
195  putText(frame, label, cv::Point(0, 15), cv::FONT_HERSHEY_SIMPLEX, 0.5,
196  cv::Scalar(0, 0, 255));
197 
198  // Write the frame with the detection boxes
199  cv::Mat detectedFrame;
200  frame.convertTo(detectedFrame, CV_8U);
201  if (parser.has("image"))
202  input.imgProcessor('w', detectedFrame);
203  else
204  input.videoProcessor('w', detectedFrame, video);
205 
206  static const std::string Window_name = " Human detection";
207  namedWindow(Window_name, cv::WINDOW_NORMAL);
208  imshow(Window_name, frame);
209  if (parser.has("image")) {
210  cv::waitKey(3000);
211  }
212  }
213 
214  cap.release();
215  if (parser.has("video")) video.release();
216 }
